```rust
use bevy::prelude::*;

#[derive(Component)]
struct DialogBox;

fn main() {
    App::new()
        .add_plugins(DefaultPlugins)
        .add_systems(Startup, setup)
        .add_systems(Update, update_dialog_size)
        .run();
}

fn setup(mut commands: Commands, asset_server: Res<AssetServer>) {
    // 创建2D相机
    commands.spawn(Camera2d);
    
    // 创建ImageNode显示图像
    commands.spawn((
        Node {
            position_type: PositionType::Absolute,
            bottom: Val::Px(50.0),
            left: Val::Px(0.0),
            width: Val::Px(1920.0),
            height: Val::Px(270.0),
            ..default()
        },
        ImageNode::new(asset_server.load("textbox.png")),
        ZIndex(0),
        DialogBox,
    ));
}

fn update_dialog_size(
    mut dialog_query: Query<&mut Node, With<DialogBox>>,
    windows: Query<&Window>,
) {
    if let (Ok(mut node), Ok(window)) = (dialog_query.single_mut(), windows.single()) {
        let window_width = window.width();
        let window_height = window.height();
        
        // 保持原始比例 1920:270 = 7.11:1
        let target_width = window_width;
        let target_height = window_width / 7.11;
        
        // 确保不超出窗口高度的30%
        let max_height = window_height * 0.3;
        let final_height = target_height.min(max_height);
        let final_width = if target_height > max_height {
            max_height * 7.11
        } else {
            target_width
        };
        
        node.width = Val::Px(final_width);
        node.height = Val::Px(final_height);
        node.bottom = Val::Px(window_height * 0.05); // 距底部5%
    }
}
```
