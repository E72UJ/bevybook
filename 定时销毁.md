```rust
use bevy::prelude::*;
use std::time::Duration;

#[derive(Component)]
struct DestroyAfter {
    timer: Timer,
}

impl DestroyAfter {
    fn new(duration: Duration) -> Self {
        Self {
            timer: Timer::new(duration, TimerMode::Once),
        }
    }
}

fn main() {
    App::new()
        .add_plugins(DefaultPlugins)
        .add_systems(Startup, setup)
        .add_systems(Update, destroy_entities)
        .run();
}

fn setup(mut commands: Commands, asset_server: Res<AssetServer>) {
    // 创建2D相机
    commands.spawn(Camera2d);
    
    // 创建ImageNode，5秒后销毁
    commands.spawn((
        ImageNode::new(asset_server.load("1.png")),
        Transform::default(),
        Visibility::Visible,
        DestroyAfter::new(Duration::from_secs(5)),
    ));

    commands.spawn((
        ImageNode::new(asset_server.load("2.png")),
        Transform::default(),
        Visibility::Visible,
        DestroyAfter::new(Duration::from_secs(10)),
    ));
}

fn destroy_entities(
    mut commands: Commands,
    mut query: Query<(Entity, &mut DestroyAfter)>,
    time: Res<Time>,
) {
    for (entity, mut destroy_after) in query.iter_mut() {
        destroy_after.timer.tick(time.delta());
        if destroy_after.timer.finished() {
            commands.entity(entity).despawn();
        }
    }
}
```
